# RLM × PEEK v5 — Decisions Distiller + Split-Serial-Parallel Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to execute task-by-task.

**Goal:** Recover v3's `decision_logic` 81% (v4 got 72%) by tuning PEEK's distiller to extract canonical decisions into REUSABLE RESULTS, AND cut wall-clock from 6h to ~3h by splitting the run: serial first 30 questions to build the map (PEEK live), then parallel remaining 62 (skip the 2 known-timeout items) with the frozen map injected as a static string into the system prompt.

**Architecture:** Two-phase run sharing one rubrics-v5 JSON via the existing `--jsonl` resume. Phase 5a uses the existing serial-PEEK loop with a new `--peek-distiller-addendum-preset decisions` flag. Phase 5b loads the frozen map TEXT from the PEEK JSON, injects it via `PAPERS_QA_SYSTEM_PROMPT_ADDENDUM` env var, and runs through the existing `run_inference` with workers=4 (no live PEEK — the map is now just frozen prompt text).

**Tech Stack:** Python 3.12, existing peek-ai + papers_qa, no new deps.

```python
def build_peek_policy(
    cfg: PeekCfg,
    *,
    client: LMClient | None = None,
) -> CachePolicy:
    if client is None:
        api_key = cfg.api_key or os.environ.get("OPENAI_API_KEY")
        if not api_key:
            raise ValueError("build_peek_policy: no client passed and OPENAI_API_KEY not set")
        base_url = cfg.base_url or os.environ.get("OPENAI_BASE_URL")
        client = OpenAIClient(model=cfg.distiller_model, api_key=api_key, base_url=base_url)
    policy = CachePolicy(
        client=client,
        token_budget=cfg.token_budget,
        evolve_steps=cfg.evolve_steps,
    )
    if cfg.distiller_addendum is not None:
        from peek.core.distiller import Distiller
        custom_prompt = policy._distiller.prompt + cfg.distiller_addendum
        policy._distiller = Distiller(policy.client, prompt=custom_prompt)
    return policy
```

```python
def _parse_item_filter(spec: str | None, *, mode: str) -> set[int] | None:
    """Parse --include-items or --skip-items spec strings.

    include modes (e.g., '0-29' or '30-93,7' or '5'): returns set of allowed item_idxs.
    skip modes (CSV like '46,48'): returns set of disallowed item_idxs.
    """
    if spec is None:
        return None
    out: set[int] = set()
    for part in spec.split(","):
        part = part.strip()
        if not part:
            continue
        if "-" in part:
            lo, hi = part.split("-", 1)
            out.update(range(int(lo), int(hi) + 1))
        else:
            out.add(int(part))
    return out


def _frozen_map_to_addendum(map_path: Path) -> str:
    """Read the PEEK map JSON, brace-escape the map_text, wrap in delimiters."""
    payload = json.loads(map_path.read_text(encoding="utf-8"))
    map_text = payload.get("map_text", "")
    escaped = map_text.replace("{", "{{").replace("}", "}}")
    return (
        "\n================ Context Map (PEEK frozen) ================\n"
        + escaped
        + "\n================ End of Context Map ================\n"
    )
```

```bash
import json
from collections import Counter

def mean_anc(s):
    xs = [r["score_anchored"]["normalized"] for r in s
          if r.get("score_anchored") and r["score_anchored"].get("normalized") is not None]
    return sum(xs)/len(xs) if xs else 0
def ct_rates(s):
    counter = {}
    for r in s:
        for b in r.get("breakdown",[]):
            if b["sign"] != "positive": continue
            ct = b["criterion_type"]; slot = counter.setdefault(ct, [0,0])
            slot[0] += 1
            if b["met"]: slot[1] += 1
    return {ct: m/t*100 if t else 0 for ct,(t,m) in counter.items()}
def trips(s, key):
    return sum(1 for r in s for b in r.get("breakdown",[]) if b["criterion_type"]=="anti_hacking" and b["met"] and key in b["text"])

s = {v: json.load(open(f"outputs/scoring/cae-v2.0-1-scores{'-' + v if v != 'v1' else ''}.json"))
     for v in ["v1", "v3", "v4", "v5"]}

print(f"=== anchored means ===")
for v in ["v1", "v3", "v4", "v5"]:
    print(f"  {v}: {mean_anc(s[v]):.3f}  ({len(s[v])} items)")
print(f"\n  AC #v5 (v5 anchored ≥ v3 − 0.02): {'PASS ✅' if mean_anc(s['v5']) >= mean_anc(s['v3']) - 0.02 else 'FAIL ❌'}")

print(f"\n=== decision_logic (target: recover v1's 81%) ===")
for v in ["v1", "v3", "v4", "v5"]:
    print(f"  {v}: {ct_rates(s[v]).get('decision_logic',0):.1f}%")

print(f"\n=== all criterion types ===")
all_cts = set()
for v in s.values(): all_cts.update(ct_rates(v).keys())
for ct in sorted(all_cts):
    print(f"  {ct:25s}  v1={ct_rates(s['v1']).get(ct,0):5.1f}%  v3={ct_rates(s['v3']).get(ct,0):5.1f}%  v4={ct_rates(s['v4']).get(ct,0):5.1f}%  v5={ct_rates(s['v5']).get(ct,0):5.1f}%")

print(f"\n=== pitfalls ===")
for name, key in [("verbosity", "回答篇幅冗长"), ("boilerplate", "回答以套话")]:
    print(f"  {name}: v1={trips(s['v1'],key)} v3={trips(s['v3'],key)} v4={trips(s['v4'],key)} v5={trips(s['v5'],key)}")
PY
```
