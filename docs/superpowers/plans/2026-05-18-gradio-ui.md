**Goal:** Add a Gradio web UI to `papers_qa` that streams per-iteration reasoning trajectory live, supports stateless multi-turn chat, and can be shared publicly via `share=True`.

**Architecture:** A new `papers_qa/streaming.py` module wraps the existing `PapersQA` driver with a `_QueueLogger` (subclass of `rlm.logger.RLMLogger`) that pushes each completed `RLMIteration` onto a `queue.Queue`. - `gradio>=4.44.0` (new optional dep)
- Existing `rlms`, `python-dotenv`
- `queue.Queue` + `threading.Thread` for stream bridging

```python
"""Live event stream from RLM iterations, for the Gradio UI.

We subclass ``rlm.logger.RLMLogger`` so that every completed ``RLMIteration``
is pushed onto a ``queue.Queue`` *in addition to* being written to JSONL on
disk. ``stream_ask`` runs ``rlm.completion`` on a daemon thread and yields
events from the queue, so the caller (Gradio's generator-streaming) can
update the UI in real time.

Public API:
- ``stream_ask(qa, question, history_text="") -> Iterator[dict]``
"""
from __future__ import annotations

import logging
import queue
import threading
from collections.abc import Iterator
from typing import Any

from rlm.core.types import RLMIteration
from rlm.logger import RLMLogger

from papers_qa.runner import PapersQA

logger = logging.getLogger(__name__)

# Sentinel used internally to signal "background thread finished."
_DONE_SENTINEL = {"__type__": "_done"}

# How long to wait between events before giving up. Tuned for the worst-case
# single-iteration latency we've observed on aiberm.com (≈30 s) with a wide
# margin for upstream hiccups.
_EVENT_TIMEOUT_S = 600.0

# Truncate prior assistant turns when stitching history into the new
# root_prompt — keeps the root LLM's context bounded.
_HISTORY_TURN_TRUNCATE = 800


class _QueueLogger(RLMLogger):
    """RLMLogger that also pushes structured iteration events onto a Queue."""

    def __init__(self, log_dir: str | None, events: "queue.Queue[dict[str, Any]]") -> None:
        super().__init__(log_dir=log_dir)
        self._events = events

    def log(self, iteration: RLMIteration) -> None:  # type: ignore[override]
        super().log(iteration)
        try:
            self._events.put(_iteration_to_event(iteration))
        except Exception:
            logger.exception("Failed to push iteration event")


def _iteration_to_event(it: RLMIteration) -> dict[str, Any]:
    """Convert an RLMIteration into a JSON-serializable event dict."""
    code_blocks = []
    for cb in it.code_blocks:
        result = cb.result
        code_blocks.append(
            {
                "code": cb.code,
                "stdout": result.stdout if result else "",
                "stderr": result.stderr if result else "",
                "final_answer": getattr(result, "final_answer", None) if result else None,
            }
        )
    return {
        "type": "iteration",
        "response": it.response or "",
        "code_blocks": code_blocks,
        "final_answer": it.final_answer,
        "iteration_time": it.iteration_time,
    }


def _build_root_prompt(question: str, history_text: str) -> str:
    """Prepend a textual chat history block to the user's question."""
    if not history_text.strip():
        return question
    return (
        "对话历史（参考用，不要重复回答已答过的部分）：\n"
        f"{history_text}\n\n新问题：{question}"
    )


def stream_ask(
    qa: PapersQA,
    question: str,
    history_text: str = "",
) -> Iterator[dict[str, Any]]:
    """Yield iteration events as RLM thinks, then a final or error event.

    Args:
        qa: A live PapersQA instance (papers + tools + RLM already constructed).
        question: The user's question (typically Chinese).
        history_text: Optional textual chat history to prepend. Empty disables.

    Yields:
        ``{"type": "iteration", "response": str, "code_blocks": [...],
           "final_answer": str | None, "iteration_time": float}``
        ``{"type": "final", "answer": str, "duration_s": float,
           "cost_usd": float | None}`` once successful
        ``{"type": "error", "error": "TypeName: msg"}`` on exception
        ``{"type": "timeout"}`` if no event for ``_EVENT_TIMEOUT_S`` seconds
    """
    events: "queue.Queue[dict[str, Any]]" = queue.Queue()

    original_logger = qa.rlm.logger
    streaming_logger = _QueueLogger(
        log_dir=str(qa.config.log_dir),
        events=events,
    )
    qa.rlm.logger = streaming_logger

    root_prompt = _build_root_prompt(question, history_text)
    result_holder: dict[str, Any] = {"result": None, "error": None}

    def run() -> None:
        try:
            result_holder["result"] = qa.rlm.completion(
                prompt=qa.papers,
                root_prompt=root_prompt,
            )
        except Exception as e:  # noqa: BLE001 — surface everything to UI
            result_holder["error"] = e
        finally:
            events.put(_DONE_SENTINEL)

    thread = threading.Thread(target=run, daemon=True)
    thread.start()

    try:
        while True:
            try:
                evt = events.get(timeout=_EVENT_TIMEOUT_S)
            except queue.Empty:
                yield {"type": "timeout"}
                return
            if evt is _DONE_SENTINEL or evt.get("__type__") == "_done":
                if result_holder["error"] is not None:
                    err = result_holder["error"]
                    yield {"type": "error", "error": f"{type(err).__name__}: {err}"}
                    return
                r = result_holder["result"]
                cost = (
                    r.usage_summary.total_cost
                    if r.usage_summary is not None
                    else None
                )
                yield {
                    "type": "final",
                    "answer": r.response,
                    "duration_s": r.execution_time,
                    "cost_usd": cost,
                }
                return
            yield evt
    finally:
        qa.rlm.logger = original_logger


def build_history_text(history: list[tuple[str, str]]) -> str:
    """Format a list of (user, bot) turn pairs into a single history block.

    Truncates each assistant message to ``_HISTORY_TURN_TRUNCATE`` chars to
    keep the root prompt bounded across long conversations.
    """
    if not history:
        return ""
    lines: list[str] = []
    for q, a in history:
        if not q and not a:
            continue
        if q:
            lines.append(f"Q: {q}")
        if a:
            snippet = a if len(a) <= _HISTORY_TURN_TRUNCATE else a[:_HISTORY_TURN_TRUNCATE] + "…"
            lines.append(f"A: {snippet}")
    return "\n".join(lines)
```

_RESPONSE_TRUNCATE = 600
_STDOUT_TRUNCATE = 1500
_STDERR_TRUNCATE = 500


def _truncate(s: str, n: int) -> str:
    if len(s) <= n:
        return s
    return s[:n] + "\n… (truncated)"


def render_iteration_md(n: int, evt: dict[str, Any]) -> str:
    """Render one iteration event as a collapsible Markdown card."""
    parts: list[str] = []
    parts.append(f"\n### 🔹 迭代 {n} · {evt.get('iteration_time', 0):.1f}s\n")
    resp = (evt.get("response") or "").strip()
    if resp:
        parts.append(f"**模型输出：**\n\n```\n{_truncate(resp, _RESPONSE_TRUNCATE)}\n```\n")
    for i, cb in enumerate(evt.get("code_blocks") or [], 1):
        parts.append(f"<details open><summary>📦 代码块 {i}</summary>\n\n")
        parts.append(f"```python\n{cb['code']}\n```\n\n")
        if cb.get("stdout"):
            parts.append(f"**stdout：**\n\n```\n{_truncate(cb['stdout'], _STDOUT_TRUNCATE)}\n```\n\n")
        if cb.get("stderr"):
            parts.append(f"**stderr：**\n\n```\n{_truncate(cb['stderr'], _STDERR_TRUNCATE)}\n```\n\n")
        if cb.get("final_answer"):
            parts.append("✅ **此块已置 `answer.ready = True`**\n\n")
        parts.append("</details>\n\n")
    return "".join(parts)


def render_error_md(msg: str) -> str:
    return f"\n### ❌ 出错\n\n```\n{msg}\n```\n"


def render_timeout_md() -> str:
    return "\n### ⚠️ 模型 600s 无响应\n\n请重试，或检查 API 端点状态。\n"

def make_chat_handler(qa: PapersQA):
    """Return a Gradio generator handler closed over the shared PapersQA."""

    def handler(
        message: str,
        history: list[tuple[str, str]],
    ) -> Iterator[tuple[list[tuple[str, str]], str, str]]:
        if not message.strip():
            yield history, "", "请先输入问题。"
            return

        prior_history_text = build_history_text(history)

        # Show user turn immediately with an empty bot slot
        history_pending = history + [(message, "")]
        trace_md = ""
        t0 = time.perf_counter()
        status = f"⏱ 0.0s · 初始化…"
        yield history_pending, trace_md, status

        iter_count = 0
        final_answer: str | None = None

        try:
            for evt in stream_ask(qa, message, history_text=prior_history_text):
                etype = evt["type"]
                if etype == "iteration":
                    iter_count += 1
                    trace_md += render_iteration_md(iter_count, evt)
                    elapsed = time.perf_counter() - t0
                    status = f"⏱ {elapsed:.1f}s · 迭代 {iter_count} · 思考中…"
                    yield history_pending, trace_md, status
                elif etype == "final":
                    final_answer = evt["answer"]
                    elapsed = evt.get("duration_s", time.perf_counter() - t0)
                    cost = evt.get("cost_usd")
                    cost_str = f" · ${cost:.4f}" if cost else ""
                    status = f"✅ 完成 · {elapsed:.1f}s · 迭代 {iter_count}{cost_str}"
                    final_history = history + [(message, final_answer)]
                    yield final_history, trace_md, status
                    return
                elif etype == "error":
                    trace_md += render_error_md(evt["error"])
                    elapsed = time.perf_counter() - t0
                    status = f"❌ 失败 · {elapsed:.1f}s"
                    err_history = history + [(message, f"❌ 出错：{evt['error']}")]
                    yield err_history, trace_md, status
                    return
                elif etype == "timeout":
                    trace_md += render_timeout_md()
                    elapsed = time.perf_counter() - t0
                    status = f"⏰ 超时 · {elapsed:.1f}s"
                    to_history = history + [(message, "⚠️ 模型 600s 无响应，请重试")]
                    yield to_history, trace_md, status
                    return
        except Exception as e:  # noqa: BLE001
            log.exception("Unexpected error in chat handler")
            trace_md += render_error_md(f"{type(e).__name__}: {e}")
            elapsed = time.perf_counter() - t0
            status = f"❌ 失败 · {elapsed:.1f}s"
            err_history = history + [(message, f"❌ 出错：{type(e).__name__}: {e}")]
            yield err_history, trace_md, status

    return handler

def _env_bool(key: str, default: bool = False) -> bool:
    val = os.environ.get(key, "").lower()
    if val in ("true", "1", "yes", "on"):
        return True
    if val in ("false", "0", "no", "off"):
        return False
    return default


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="Gradio web UI for papers_qa.")
    parser.add_argument("--share", action="store_true", help="Publish a public *.gradio.live link.")
    parser.add_argument("--no-share", action="store_true", help="Force-disable share (overrides env).")
    parser.add_argument("--port", type=int, default=None)
    parser.add_argument("--host", type=str, default=None)
    parser.add_argument("--title", type=str, default="papers_qa")
    parser.add_argument("--env-file", type=str, default=None)
    parser.add_argument("--log-level", default="INFO", choices=["DEBUG", "INFO", "WARNING", "ERROR"])
    return parser.parse_args()


def main() -> int:
    args = parse_args()
    logging.basicConfig(
        level=getattr(logging, args.log_level),
        format="%(asctime)s %(name)s %(levelname)s: %(message)s",
    )

    cfg = PapersQAConfig.from_env(args.env_file)
    log.info("Booting Gradio app…")
    qa = PapersQA(cfg)
    demo = build_demo(qa, banner_title=args.title)
    demo.queue(default_concurrency_limit=2, max_size=20)

    if args.no_share:
        share = False
    elif args.share:
        share = True
    else:
        share = _env_bool("PAPERS_QA_GRADIO_SHARE", default=False)

    port = args.port or int(os.environ.get("PAPERS_QA_GRADIO_PORT", "7860"))
    host = args.host or os.environ.get("PAPERS_QA_GRADIO_HOST", "127.0.0.1")

    log.info("Launching: share=%s host=%s port=%s", share, host, port)
    demo.launch(
        share=share,
        server_port=port,
        server_name=host,
        show_error=True,
    )
    return 0


if __name__ == "__main__":
    sys.exit(main())

## Risks

| Severity | Risk | Mitigation |
|----------|------|------------|
| HIGH | Public `--share` link burns API budget if leaked | `default_concurrency_limit=2`, `max_size=20`, per-request budget cap via `PAPERS_QA_MAX_BUDGET_USD`. Documented in README and 中文文档. |
| MEDIUM | Concurrent requests interleave logger writes | Gradio queue serializes to 2 at a time. Worst case: trace lines appear under the wrong session for a moment. Acceptable for v1; v2 fix is per-request RLM. |
| MEDIUM | Long stdout/stderr from REPL chokes browser | Truncate to 1500/500 chars in `render_iteration_md`. |
| LOW | Browser tab kept open during a 30+ iteration run | Trace panel has `max-height: 640px` + scroll. |
| LOW | Gradio version drift breaks API (e.g., `bubble_full_width` deprecated) | Pin `gradio>=4.44.0` floor; if a future major release breaks, version-bump and adjust. |
